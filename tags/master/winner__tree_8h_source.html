27 /*!
28  * The class winner_tree is a binary tournament tree. There are n=2^k so
29  * called players which compete for the winning position. The winner is the
30  * smallest one regarding the comparator m_less. Each player is identified with
31  * it's index. The comparator should use this index in order to compare the
32  * actual values.
33  *
34  * A change of one players value results in O(log n) operations. The winner
35  * can be determined in O(1) operations.
36  *
37  * There is a number of fixed players which remain at the same index forever,
38  * even if they were deactivated in between. Additionally any number of players
39  * can be added dynamically. They are assigned to a free slot and if one gets
40  * removed the slot will be marked free again.
41  */
42 template <typename Comparator>
43 class winner_tree

78  * Constructor. Reserves space, registers free slots. No games are played

84  * \param less The comparator. It should use two given indices, compare the

127  /**
128  * deactivate one player in a batch of deactivations run
129  * replay_on_deactivation() afterwards. If there are multiple consecutive
130  * removes you should run deactivate_player_step() for all of them first and
131  * afterwards run replay_on_deactivation() for each one of them.
132  */
133  inline void deactivate_player_step(unsigned int index)
134  {
135  m_stats.remove_player_time.start();
136  assert(index < m_num_slots);
137  int p = (int)(m_tree.size() / 2) + index;
138 
139  // Needed for following deactivations...
140  while (p >= 0 && m_tree[p] == (int)index) {
141  m_tree[p] = -1;
142  p -= (p + 1) % 2;
143  p /= 2;
144  }
145 
146  m_stats.remove_player_time.stop();

187  /**
188  * Replays all games the player with the given index is involved in. This
189  * corresponds to the path from leaf [index] to root. If only the value of
190  * player [index] has changed the result is a valid winner tree.
191  *
192  * \param index The player whose value has changed.
193  *
194  * \param done Set done to true if the player has been deactivated
195  * or removed. All games will be lost then.
196  */
197  inline void replay_on_change(unsigned int index, bool done = false)
198  {
199  m_stats.replay_time.start();
200 
201  int top;
202  int p = static_cast<int>((m_tree.size() / 2) + index);
203 
204  if (!done) {
205  top = index;
206  }
207  else {
208  top = -1;
209  }
210 
211  while (p > 0) {
212  m_tree[p] = top;
213  p -= (p + 1) % 2; // round down to left node position
214 
215  if (m_tree[p] < 0)
216  top = m_tree[p + 1];
217  else if (m_tree[p + 1] < 0)
218  top = m_tree[p];
219  else if (m_less(m_tree[p], m_tree[p + 1]))
220  top = m_tree[p];
221  else
222  top = m_tree[p + 1];
223 
224  p /= 2;
225  }
226 
227  m_tree[p] = top;
228 
229  m_stats.replay_time.stop();
230  }

232  //! Doubles the number of slots and adapts the tree so it's a valid winner
233  //! tree again.
234  inline void double_num_slots()
235  {
236  m_stats.double_num_slots_time.start();
237 
238  m_num_slots = m_num_slots << 1;
239  unsigned old_tree_size = static_cast<unsigned>(m_tree.size());
240  unsigned tree_size = (m_num_slots << 1) - 1;
241  m_tree.resize(tree_size, -1);
242 
243  for (int i = old_tree_size - 1; i >= 0; --i) {
244  unsigned old_index = i;
245  unsigned old_level = ilog2_floor(old_index + 1);
246  unsigned new_index = old_index + (1 << old_level);
247  m_tree[new_index] = m_tree[old_index];
248  }
249 
250  unsigned step_size = (1 << ilog2_floor(old_tree_size));
251  unsigned index = tree_size - 1;
252 
253  while (step_size > 0) {
254  for (unsigned i = 0; i < step_size; ++i) {
255  m_tree[index] = -1;
256  --index;
257  }
258  index -= step_size;
259  step_size /= 2;
260  }
261 
262  m_stats.double_num_slots_time.stop();
263  }

296  //! Build from winner tree from scratch.
297  inline void rebuild()
298  {
299  for (int i = (int)(m_tree.size() / 2) - 1; i >= 0; --i) {
300  const int lc = i * 2 + 1;
301  const int rc = i * 2 + 2;
302  int winner;
303 
304  if (m_tree[lc] < 0)
305  winner = m_tree[rc];
306  else if (m_tree[rc] < 0)
307  winner = m_tree[lc];
308  else if (m_less(m_tree[lc], m_tree[rc]))
309  winner = m_tree[lc];
310  else
311  winner = m_tree[rc];
312 
313  m_tree[i] = winner;
314  }
315  }
